Étape 11 – Finalisation et optimisation côté client

Nous finalisons l’application côté client : interface intuitive, validation, affichage en temps réel, et compatibilité mobile. 


// public/js/script.js
// Gestion de la création, affichage et suppression côté client
async function supprimerIdee(id){
  await fetch("/api/idees/" + id, {
    method: "DELETE",
    headers: { Authorization: "Bearer " + localStorage.getItem("token") }
  });
  chargerIdees();
}

// Optimisation : éviter rechargement complet, améliorer accessibilité
listeIdees.addEventListener("click", (e) => {
  if(e.target.classList.contains("deleteBtn")){
    supprimerIdee(e.target.dataset.id);
  }
});

  • Affichage temps réel et suppression
  • Compatibilité mobile et accessibilité
