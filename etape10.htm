Étape 10 – Styles et accessibilité mobile-first

Nous allons créer un CSS commun pour toutes les pages, responsive et accessible, en priorité mobile-first. 


/* public/css/styles.css */
body { font-family: Arial, sans-serif; margin:0; padding:0; line-height:1.5; }
.navbar { display:flex; justify-content:space-between; align-items:center; padding:1rem; background:#333; color:#fff; }
.nav-links { display:flex; gap:1rem; list-style:none; }
.nav-links a { color:#fff; text-decoration:none; }
.nav-toggle { display:none; background:none; border:none; color:#fff; font-size:1.5rem; }
.idee { border:1px solid #ccc; padding:1rem; margin:0.5rem 0; }
img { max-width:100%; height:auto; }

@media(max-width:768px){
  .nav-links { flex-direction:column; display:none; }
  .nav-links.active { display:flex; }
  .nav-toggle { display:block; }
}


// public/js/script.js
const navToggle = document.getElementById("nav-toggle");
const navLinks = document.getElementById("nav-links");
navToggle.addEventListener("click", () => {
  navLinks.classList.toggle("active");
});
